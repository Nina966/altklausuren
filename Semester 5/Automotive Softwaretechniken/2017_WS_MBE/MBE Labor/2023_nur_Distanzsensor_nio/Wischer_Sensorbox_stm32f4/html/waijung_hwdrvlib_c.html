506{

514int SysTimer_IsTimeout(SYS_TIMER_STRUCT* timer)



515{



516  uint32_t Capture;



517



518  /* Capture current tick */



519  Capture = (uint32_t)systick_count;



520



521  /* Check */



522  if (Capture >= timer->start) {



523    if ((Capture - timer->start) >= timer->length)



524      return 1;



525  } else {



526    if (((0xFFFFFFFF - timer->start)+Capture) >= timer->length)



527      return 1;



528  }



529



530  return 0;

536void SysTimer_uS_Start(SYS_TIMER_uS_STRUCT *timer, uint32_t us)



537{



538  /* Tick */



539  timer->tick = SYS_CURRENT_TICK;



540



541  /* Tick overflow count */



542  timer->tid = (uint32_t)systick_count;



543

548int SysTimer_uS_IsTimeout(SYS_TIMER_uS_STRUCT* timer)



549{



550  uint32_t curr_tick;



551  uint32_t curr_tid;



552



553  /* Already timeout ? */



554  if (timer->target_us == 0) {



555    return 1;



556  }



557



558  /* Capture current tick */



559  curr_tick = SYS_CURRENT_TICK;



560  curr_tid = systick_count;



561



562  /* Processing TID*/



563  while (timer->tid != curr_tid) {



564    if (timer->target_us > timer->tick) {



565      timer->target_us -= timer->tick;



566      timer->tid ++;



567      timer->tick = SYSTICKRELOADVALUE-1;



568    } else {



569      timer->target_us = 0;



570      return 1;                        /* Timeout */



571    }



572  }



573



574  /* Processing Tick */



575  if ((timer->tick > curr_tick) && ((timer->tick - curr_tick) > timer->target_us))



576  {



577    timer->target_us = 0;



578    return 1;                          /* Timeout */



579  }



580



581  /* Need wait */



582  return 0;



583}

780  }
